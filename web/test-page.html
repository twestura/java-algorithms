public static void binarySearch(int[] a, int key) {
    int left = 0, right = a.length;
    // a[..left] < key, a[right..] > key
    while (left < right) {
        var mid = (left + right) >>> 1;
        var x = a[mid];
        if (x == key) return mid;
        if (x < key) left = mid + 1;
        else right = mid;
    }
    return -1;
}

For inline code, such as int x = 5, we just use a code tag without a pre tag. 

/** Merges a[left..mid] and a[mid..right]. */
public static void merge(int[] a, int left, int mid, int right, int[] t) {
    var leftLen = mid - left;
    System.arraycopy(a, left, t, 0, leftLen);
    int i = 0, j = mid, k = left;
    while (i < leftLen && j < right)
        a[k++] = a[j] > t[i] ? t[i++] : a[j++];
    System.arrayCopy(t, i, a, k, leftLen - i);
}
